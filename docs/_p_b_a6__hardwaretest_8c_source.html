378 inline void ISR_UartRx(void)
379 {
380  static uint8_t len=0; /*Anzahl empfangene Zeichen*/
381  uint8_t c; /*Aktuell eingelesenes Zeichen*/
382  while(PIR1bits.RCIF) /*Zeichen im Buffer*/
383  {
384  c=UART_Getc(); /*Zeichen einlesen*/
385  if(8==c) /*Wenn ein Backspace empfangen wurde*/
386  {
387  if(len>0) /*Wenn schon Zeichen empfangen wurden*/
388  {
389  len--; /*Zeichen dekrementieren*/
390  UART_Putc(c); /*Backspace zur√ºcksenden*/
391  UART_Putc(' '); /*gel√∂schtes Zeichen √ºberschreiben*/
392  UART_Putc(c); /*Backspace zur√ºcksenden*/
393  }
394  }
395  else if (13==c) /*Carriage Return empfangen*/
396  {
397  uartBuffer_g[len]=0; /*Nullterminierung*/
398  len=0; /*Z√§hler zur√ºcksetzen*/
399  uartHasReceived_g=1; /*Flag setzen, Kompletter Text empfangen*/
400  }
401  else if ((c>=32)&&(c<=126)) /*Wenn druckbares Zeichen empfangen wurde*/
402  {
403  if(len<(UART_BUFFERSIZE-1)) /*Wenn maximale Stringl√§nge noch nicht erreicht*/
404  {
405  uartBuffer_g[len++]=c; /*Empfangenes Zeichen in String ablegen*/
406  UART_Putc(c); /*Empfangenes Zeichen zur√ºcksenden*/
407  }
408  }
409  }
