49 void LCD_Send(uint8_t value)
50 {
51  LCD_EN=1; /*Pos. Flanke am Enable-Pin*/
52  LCD_D7 = BIT_TEST(value,7); /*Oberes Daten-Nibble anlegen*/
53  LCD_D6 = BIT_TEST(value,6);
54  LCD_D5 = BIT_TEST(value,5);
55  LCD_D4 = BIT_TEST(value,4);
56  LCD_EN=0; /*Oberes Nibble senden*/
57  #if (_XTAL_FREQ>20000000)
58  NOP(); /*1 Zyklus warten-> verhindert zu kurzen Puls bei 32MHz (min. 130ns stable)*/
59  #endif
60  LCD_EN=1; /*Pos. Flanke am Enable-Pin*/
61  LCD_D7 = BIT_TEST(value,3); /*Unteres Daten-Nibble anlegen*/
62  LCD_D6 = BIT_TEST(value,2);
63  LCD_D5 = BIT_TEST(value,1);
64  LCD_D4 = BIT_TEST(value,0);
65  LCD_EN=0; /*Unteres Nibble senden*/
66  __delay_us(30); /*Maximaler Zeitbedarf einer Übertragung (blocking)*/
67  }

157 void LCD_Putc(uint8_t ch)

164  default:
165  LCD_SEND_DATA(LCD_GetSpecChar(ch)); /*Zeichen an LCD senden*/
