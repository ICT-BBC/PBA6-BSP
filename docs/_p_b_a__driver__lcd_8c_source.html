51 void LCD_Send(uint8_t value)
52 {
53  LCD_EN=1; /*Pos. Flanke am Enable-Pin*/
54  LCD_D7 = BIT_TEST(value,7); /*Oberes Daten-Nibble anlegen*/
55  LCD_D6 = BIT_TEST(value,6);
56  LCD_D5 = BIT_TEST(value,5);
57  LCD_D4 = BIT_TEST(value,4);
58  LCD_EN=0; /*Oberes Nibble senden*/
59  #if (_XTAL_FREQ>20000000)
60  NOP(); /*1 Zyklus warten-> verhindert zu kurzen Puls bei 32MHz (min. 130ns stable)*/
61  #endif
62  LCD_EN=1; /*Pos. Flanke am Enable-Pin*/
63  LCD_D7 = BIT_TEST(value,3); /*Unteres Daten-Nibble anlegen*/
64  LCD_D6 = BIT_TEST(value,2);
65  LCD_D5 = BIT_TEST(value,1);
66  LCD_D4 = BIT_TEST(value,0);
67  LCD_EN=0; /*Unteres Nibble senden*/
68  __delay_us(30); /*Maximaler Zeitbedarf einer Übertragung (blocking)*/
69  }
