108 {
109  static uint8_t switchFlags = 0;
110  static uint8_t switchCount[8]={0};
111  uint16_t oneSec=1000/loopDelayMS;
112  uint16_t o_5Sec=500/loopDelayMS;
113  uint8_t i=0;
114  uint8_t switchValue;
115  switchValue = PORTB;
116  p_toEvents->valueEdge = ((~switchValue&switchFlags)<<8)|(switchValue&~switchFlags);
117  switchFlags = switchValue;
118  for(i=0;i<8;i++)
119  {
120  if(BIT_TEST(switchValue,i))
121  {
122  if(switchCount[i]<255)
123  {
124  switchCount[i]++;
125  if(switchCount[i]>=oneSec) /*1 Sekunde Taster gedrückt*/
126  {
127  p_toEvents->valuePressed|=((uint16_t)1<<(8+i));//Bit setzen
128  }
129 
130  else
131  {
132  if(switchCount[i]>=o_5Sec) /*0.5 Sekunden Taster gedrückt*/
133  {
134  p_toEvents->valuePressed|=((uint16_t)1<<i); /*Bit setzen*/
135  }
136  }
137  }
138  }

140  {
141  switchCount[i]=0; /*Entsprechenden Zähler zurücksetzen*/

152 inline void EVENTS_TimerISR(void)
153 {
154  static uint8_t stateOld=0;
155  uint8_t i;
156  if(stateOld!=*p_toState) /*Bei Zustandswechsel alle Timeouts zurücksetzen*/
157  for(i=0;i<MAXTIMEOUTS;i++)
158  {
159  timeoutValues.eventOccured[i]=0;
160  timeoutValues.active[i]=0;
161  timeoutValues.ms[i]=0;
162  }
163  else
164  for(i=0;i<MAXTIMEOUTS;i++) /*Timoutereignisse prüfen*/
165  if((timeoutValues.active[i]==1) && (timeoutCounter==timeoutValues.end[i]))
166  timeoutValues.eventOccured[i]=1;
167  stateOld=*p_toState;
168  ++timeoutCounter;
169 }

176 static uint8_t EVENTS_TimeoutHandler(uint16_t timeoutMS)
177 {
178  if(timeoutMS<=loopDelayMS) /*Wenn gewünschtes timeout kleiner oder gleich eingestelltem Loopdelay*/
179  {
180  return 1; /*Rückgabe: Timeoutereignis eingetreten->timeout=loop_delay*/
181  }
182  else
183  {
184  uint8_t i;
185  /*Prüfen ob Timeout schon vorhanden*/
186  for(i=0; i<MAXTIMEOUTS && timeoutValues.ms[i]!=timeoutMS; i++);
187  if(i==MAXTIMEOUTS) /*Timout noch nicht vorhanden*/
188  {
189  /*Prüfen ob noch Timeouts vorhanden sind*/
190  for(i=0; i<MAXTIMEOUTS && timeoutValues.active[i]>0; i++);
191  if(i==MAXTIMEOUTS) return 0xFF; /*Rückgabe: Fehler keine Timeouts mehr verfügbar*/
192  /*neues Timeout speichern*/
193  INTCONbits.GIE=0; /*Interrupts ausschalten (counter absichern)*/
194  timeoutValues.active[i] = 1;
195  timeoutValues.eventOccured[i]=0;
196  timeoutValues.ms[i] = timeoutMS;
197  timeoutValues.end[i] = timeoutCounter+timeoutMS-2*loopDelayMS;
198  INTCONbits.GIE=1; /*Interrupts wieder einschalten*/
199  return 0; /*Rückgabe: Kein Timeoutereignis eingetreten*/
200  }
201  else /*Timeout bereits vorhanden*/
202  {
203  if(timeoutValues.active[i] && timeoutValues.eventOccured[i]) /*Timeoutereignis eingetreten*/
204  {
205  /*Timeout zurücksetzen*/
206  timeoutValues.eventOccured[i]=0;
207  timeoutValues.active[i]=0;
208  timeoutValues.ms[i]=0;
209  return 1; /*Rückgabe: Timeoutereignis eingetreten*/
210  }
211  }
212  return 0; /*Rückgabe: Kein Timeoutereignis eingetreten*/
213  }
214 }

221 static uint8_t EVENTS_ActiveUntilHandler(uint16_t activeUntilMS)
222 {
223  uint8_t i;
224  /*Prüfen ob Timeout schon vorhanden*/
225  for(i=0; i<MAXTIMEOUTS && timeoutValues.ms[i]!=activeUntilMS; i++);
226  if(i==MAXTIMEOUTS) /*Timout noch nicht vorhanden*/
227  {
228  /*Prüfen ob noch Timeouts vorhanden sind*/
229  for(i=0; i<MAXTIMEOUTS && timeoutValues.active[i]>0; i++);
230  if(i==MAXTIMEOUTS) return 0xFF; /*Rückgabe: Fehler keine Timeouts mehr verfügbar*/
231  /*neues Timeout speichern*/
232  INTCONbits.GIE=0; /*Interrupts ausschalten (counter absichern)*/
233  timeoutValues.active[i] = 1;
234  timeoutValues.eventOccured[i]=0;
235  timeoutValues.ms[i] = activeUntilMS;
236  timeoutValues.end[i] = timeoutCounter+activeUntilMS-loopDelayMS;
237  INTCONbits.GIE=1; /*Interrupts wieder einschalten*/
238  return 1; /*Rückgabe: Kein Timeoutereignis eingetreten*/
239  }
240  else /*Timeout bereits vorhanden*/
241  {
242  if(timeoutValues.active[i]==2) /*Ontime-Timemout bereits 1x abgelaufen*/
243  return 0;
244  if(timeoutValues.active[i] && timeoutValues.eventOccured[i]) /*Timeoutereignis eingetreten*/
245  {
246  /*Timeout zurücksetzen*/
247  timeoutValues.eventOccured[i]=0;
248  timeoutValues.active[i]=2;
249  return 0; /*Rückgabe: Timeoutereignis eingetreten*/
250  }
251  }
252  return 1; /*Rückgabe: Kein Timeoutereignis eingetreten*/
253 }
